// This pseudocode is intended to display employee net pay values.
// All employees have a standard $45 deduction from their checks.
// If an employee does not earn enough to cover the deduction
// an error message is displayed.
start
   Declarations
      string name
      string EOFNAME = "ZZZ"
      num hours
      num rate
      num DEDUCTION = 45
      num gross
      num net
   output "Enter first name or ZZZ to quit"
   name = input("Enter first name or ZZZ to quit")
   if name <> to EOFNAME
      output "Enter hours worked for ", name
      hours = input("Enter hours worked for ", name)
      output "Enter hourly rate for ", name
      rate = input("Enter hourly rate for ", name)
      gross = hours * rate
      net = gross - DEDUCTION
      while net > 0 then
         output "Net pay for ", name, " is ", net
      else
         output "Error: Deductions not covered. Net is 0."
      endwhile
      output "Enter first name or ZZZ to quit"
      name = input("Enter first name or ZZZ to quit")
   endif
   output "End of job"
stop
